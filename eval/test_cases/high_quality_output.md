## 1. Defining a Claude Tool with Strict Input Validation

Use this pattern when you need Claude to call external APIs or functions
with structured arguments. The JSON schema enforces types at the API level,
preventing malformed calls before they reach your code.

**Avoid:** Defining tools with `"type": "object"` but no `"required"` field —
Claude will then happily omit critical parameters, causing KeyError at runtime.

```python
import anthropic

client = anthropic.Anthropic()

# Define tool with strict schema — always include "required"
search_tool = {
    "name": "search_documents",
    "description": "Search a document corpus for relevant passages",
    "input_schema": {
        "type": "object",
        "properties": {
            "query": {
                "type": "string",
                "description": "The search query string",
            },
            "max_results": {
                "type": "integer",
                "description": "Maximum number of results to return",
                "default": 5,
            },
        },
        "required": ["query"],  # Never omit this
    },
}

response = client.messages.create(
    model="claude-opus-4-6",
    max_tokens=1024,
    tools=[search_tool],
    messages=[{"role": "user", "content": "Find documents about LoRA fine-tuning"}],
)

# Handle tool_use blocks in response
for block in response.content:
    if block.type == "tool_use":
        tool_input = block.input  # Already parsed dict
        results = search_documents(**tool_input)
```

## 2. Multi-Turn Tool Use Loop

Use this pattern for agentic workflows where Claude needs to call tools
multiple times before producing a final answer. The loop continues until
Claude returns a `stop_reason` of `"end_turn"`.

**Avoid:** Breaking out of the loop on the first `tool_use` block — some
tasks require 3-5 sequential tool calls before Claude has enough information.

```python
import anthropic
import json

client = anthropic.Anthropic()

def run_agent(user_message: str, tools: list[dict], tool_handlers: dict) -> str:
    """
    Run a multi-turn tool-use loop until Claude produces a final answer.

    Args:
        user_message:   Initial user request
        tools:          List of tool definitions
        tool_handlers:  Dict mapping tool name → callable

    Returns:
        Claude's final text response
    """
    messages = [{"role": "user", "content": user_message}]

    while True:
        response = client.messages.create(
            model="claude-opus-4-6",
            max_tokens=4096,
            tools=tools,
            messages=messages,
        )

        # Add assistant turn to message history
        messages.append({"role": "assistant", "content": response.content})

        if response.stop_reason == "end_turn":
            # Extract final text response
            return next(
                (b.text for b in response.content if hasattr(b, "text")),
                "",
            )

        if response.stop_reason != "tool_use":
            break

        # Execute all tool calls and collect results
        tool_results = []
        for block in response.content:
            if block.type != "tool_use":
                continue
            handler = tool_handlers.get(block.name)
            if not handler:
                result = {"error": f"Unknown tool: {block.name}"}
            else:
                try:
                    result = handler(**block.input)
                except Exception as e:
                    result = {"error": str(e)}

            tool_results.append({
                "type": "tool_result",
                "tool_use_id": block.id,
                "content": json.dumps(result),
            })

        messages.append({"role": "user", "content": tool_results})

    return ""
```

## 3. Streaming Responses for Long Outputs

Use streaming when generating content longer than ~500 tokens to provide
immediate user feedback. The `with` block handles connection lifecycle.

**Avoid:** Using `client.messages.create()` for long code generation tasks —
the user sees nothing until the full response completes, which can take 10-30s.

```python
import anthropic

client = anthropic.Anthropic()

def stream_code_generation(prompt: str) -> str:
    """Stream a code generation response and return the full text."""
    full_text = []

    with client.messages.stream(
        model="claude-opus-4-6",
        max_tokens=4096,
        messages=[{"role": "user", "content": prompt}],
    ) as stream:
        for text in stream.text_stream:
            print(text, end="", flush=True)
            full_text.append(text)

    print()  # Final newline
    return "".join(full_text)

# Example
result = stream_code_generation(
    "Write a Python function that implements binary search with full type hints"
)
```

## 4. Building a SKILL.md Trigger Description Generator

Use this pattern when creating new Claude skills — the description field
determines whether Claude will invoke the skill. Poor descriptions cause
the skill to never trigger or to trigger on unrelated requests.

**Avoid:** Generic descriptions like "Helps with code" — these compete with
every other skill and rarely win. Be specific about input types and output types.

```python
import anthropic
import re

client = anthropic.Anthropic()

DESCRIPTION_PROMPT = """\
You are writing the `description` field for a Claude SKILL.md file.
The description determines WHEN Claude will use this skill.

Rules for a good description:
1. Start with the primary action verb (e.g., "Generates", "Analyzes", "Converts")
2. Name the exact output format (.docx, CLAUDE.md, JSON report, etc.)
3. List 3-5 trigger phrases users would naturally say
4. Keep it under 100 words

Skill name: {name}
Skill purpose: {purpose}

Write only the description field content, no other text.
"""

def generate_skill_description(skill_name: str, skill_purpose: str) -> str:
    """Generate a high-quality trigger description for a Claude skill."""
    response = client.messages.create(
        model="claude-opus-4-6",
        max_tokens=200,
        messages=[{
            "role": "user",
            "content": DESCRIPTION_PROMPT.format(
                name=skill_name,
                purpose=skill_purpose,
            ),
        }],
    )
    return response.content[0].text.strip()

# Example
desc = generate_skill_description(
    skill_name="project-rules-generator",
```

## 5. Codebase Scanner for Rules Generation

Use this pattern as the reconnaissance phase before generating any rules file.
Reading representative source files first produces rules grounded in the actual
codebase rather than generic templates.

**Avoid:** Generating rules without scanning — you'll produce generic boilerplate
that doesn't reflect the project's actual patterns, frameworks, or conventions.

```python
import os
from pathlib import Path
from typing import Generator

# File types worth scanning (ordered by signal value)
HIGH_VALUE_EXTENSIONS = [".py", ".ts", ".tsx", ".js", ".go", ".rs"]
CONFIG_FILES = ["pyproject.toml", "package.json", "requirements.txt",
                "Dockerfile", "docker-compose.yml", ".env.example"]

def scan_codebase(root: str | Path, max_files: int = 10) -> dict:
    """
    Scan a codebase to extract patterns for rules generation.

    Returns a dict with:
        - stack:       detected languages and frameworks
        - conventions: naming and structure patterns
        - test_style:  testing framework and patterns
        - config_files: content of key config files
        - sample_code: representative source file excerpts
    """
    root = Path(root)
    result = {
        "stack": [],
        "conventions": [],
        "test_style": [],
        "config_files": {},
        "sample_code": [],
    }

    # Read config files first
    for filename in CONFIG_FILES:
        cfg_path = root / filename
        if cfg_path.exists():
            content = cfg_path.read_text(encoding="utf-8", errors="ignore")[:2000]
            result["config_files"][filename] = content

    # Detect stack from config
    if "package.json" in result["config_files"]:
        pkg = result["config_files"]["package.json"]
        if "react" in pkg.lower():
            result["stack"].append("React")
        if "typescript" in pkg.lower() or '"ts"' in pkg:
            result["stack"].append("TypeScript")
    if "pyproject.toml" in result["config_files"]:
        result["stack"].append("Python")

    # Sample source files
    sampled = 0
    for ext in HIGH_VALUE_EXTENSIONS:
        for path in root.rglob(f"*{ext}"):
            if sampled >= max_files:
                break
            # Skip test files and node_modules for the main scan
            if any(part in path.parts for part in ["node_modules", ".git", "__pycache__"]):
                continue
            try:
                content = path.read_text(encoding="utf-8", errors="ignore")[:3000]
                result["sample_code"].append({
                    "path": str(path.relative_to(root)),
                    "excerpt": content,
                })
                sampled += 1
            except OSError:
                continue

    return result
```
